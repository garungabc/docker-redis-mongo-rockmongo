docker-compose run --rm webserver-marketplace sh -c "cd /etc/nginx/certs && touch openssl.cnf && cat /etc/ssl/openssl.cnf > openssl.cnf && echo \"\" >> openssl.cnf && echo \"[ SAN ]\" >> openssl.cnf && echo \"subjectAltName=DNS.1:marketplace.local,DNS.2:*.marketplace.local\" >> openssl.cnf && openssl req -x509 -sha256 -nodes -newkey rsa:4096 -keyout marketplace.local.key -out marketplace.local.crt -days 3650 -subj \"/CN=*.marketplace.local\" -config openssl.cnf -extensions SAN && rm openssl.cnf"
